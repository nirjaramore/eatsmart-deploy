import pandas as pd
import requests
from bs4 import BeautifulSoup

class IndianFoodDataCollector:
    def __init__(self):
        self.nin_data = {}
        self.brand_data = {}
        self.fssai_data = {}
    
    # Step 1: Load ICMR-NIN published data
    def load_nin_composition_table(self):
        """
        Download from: https://www.nin.res.in/
        File: Tables on Composition of Foods Commonly Used in India (PDF)
        
        Once downloaded, parse with PyPDF2 or manual CSV creation
        """
        # Manual data from NIN tables (example)
        self.nin_data = {
            "apple": {"calories": 52, "protein": 0.26, "carbs": 13.8, "fat": 0.17},
            "rice_cooked": {"calories": 130, "protein": 2.7, "carbs": 28, "fat": 0.3},
            "dal_boiled": {"calories": 100, "protein": 8, "carbs": 18, "fat": 0.5},
            "milk": {"calories": 61, "protein": 3.2, "carbs": 4.8, "fat": 3.3},
            "curd": {"calories": 60, "protein": 3.5, "carbs": 3.1, "fat": 3},
            # ... (600+ foods from NIN tables)
        }
        return self.nin_data
    
    # Step 2: Scrape brand websites
    def scrape_britannia(self):
        """Scrape Britannia product nutrition"""
        url = "https://www.britannia.co.in/our-products"
        response = requests.get(url)
        soup = BeautifulSoup(response.content, "html.parser")
        
        # Parse products and extract nutrition info
        products = []
        for product_card in soup.find_all("div", class_="product"):
            product_name = product_card.find("h3").text
            nutrition_link = product_card.find("a", class_="nutrition")
            
            if nutrition_link:
                nutrition_page = requests.get(nutrition_link["href"])
                nutrition_soup = BeautifulSoup(nutrition_page.content, "html.parser")
                
                nutrition = self._extract_nutrition_table(nutrition_soup)
                products.append({
                    "name": product_name,
                    "brand": "Britannia",
                    "url": nutrition_link["href"],
                    **nutrition
                })
        
        return products
    
    def scrape_nestle_india(self):
        """Scrape Nestlé India products"""
        url = "https://www.nestle.in/"
        # Similar scraping logic
        pass
    
    def scrape_itc(self):
        """Scrape ITC products"""
        url = "https://www.itcportal.com/"
        # Similar scraping logic
        pass
    
    def _extract_nutrition_table(self, soup):
        """Extract nutrition facts from product page"""
        nutrition = {}
        
        # Look for nutrition table
        table = soup.find("table", class_="nutrition-facts")
        if table:
            for row in table.find_all("tr"):
                cols = row.find_all("td")
                if len(cols) >= 2:
                    nutrient_name = cols[0].text.strip().lower()
                    nutrient_value = cols[1].text.strip()
                    
                    if "calorie" in nutrient_name:
                        nutrition["calories"] = float(nutrient_value.split()[0])
                    elif "protein" in nutrient_name:
                        nutrition["protein"] = float(nutrient_value.split()[0])
                    elif "carb" in nutrient_name:
                        nutrition["carbs"] = float(nutrient_value.split()[0])
                    elif "fat" in nutrient_name:
                        nutrition["fat"] = float(nutrient_value.split()[0])
                    elif "sodium" in nutrient_name:
                        nutrition["sodium"] = float(nutrient_value.split()[0])
        
        return nutrition
    
    # Step 3: Query FSSAI for bulk data
    def request_fssai_data(self):
        """
        Email template to send to FSSAI
        """
        email_text = """
        Subject: API/Data Request for Food Product Database - Academic Research
        
        To: contact@fssai.gov.in

        I am a Computer Science student conducting research on food safety 
        and nutritional awareness among Indian consumers.

        users verify food product claims against official databases.
        
        REQUEST:
        Could you provide a list of:
        1. All approved food products in the Indian market
        2. Their registration numbers
        3. Nutrition information (if available)
        4. Safety certifications
        
        This data would be used for:
        - Academic research (final year project)
        - Building a food safety awareness app
        - Comparing claimed nutrition vs. actual values
        
        Timeline: December 2025 - May 2026 (academic year)
        Non-commercial use only
        
        The data will help create awareness about misleading food labels 
        in India.
        
        Thank you,
        [Your Name]
        [Your Email]
        [Your Phone]
        """
        
        print(email_text)
        print("\nSend this email to: contact@fssai.gov.in")
    
    # Step 4: Query Open Food Facts India
    def get_from_open_food_facts_india(self, barcode):
        """Get product data from Open Food Facts (India)"""
        url = f"https://world.openfoodfacts.org/api/v0/product/{barcode}.json?cc=in"
        
        try:
            response = requests.get(url, timeout=5)
            if response.status_code == 200:
                product = response.json()["product"]
                return {
                    "product_name": product.get("product_name"),
                    "brand": product.get("brands"),
                    "ingredients": product.get("ingredients_text"),
                    "allergens": product.get("allergens"),
                    "nutrition": product.get("nutrients"),
                    "source": "Open Food Facts India"
                }
        except Exception as e:
            print(f"Error: {e}")
        
        return None
    
    # Step 5: Combine all sources
    def get_product_comprehensive(self, product_name_or_barcode):
        """Get product info from multiple sources and combine"""
        
        data = {
            "nin_data": None,
            "brand_website": None,
            "open_food_facts": None,
            "confidence_score": 0
        }
        
        # Try NIN tables
        if product_name_or_barcode.lower() in self.nin_data:
            data["nin_data"] = self.nin_data[product_name_or_barcode.lower()]
        
        # Try Open Food Facts
        if product_name_or_barcode.isdigit():  # It's a barcode
            data["open_food_facts"] = self.get_from_open_food_facts_india(product_name_or_barcode)
        
        # Calculate confidence score
        sources_found = sum([1 for v in data.values() if v is not None])
        data["confidence_score"] = (sources_found / 3) * 100
        
        return data
