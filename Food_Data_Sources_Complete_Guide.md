### **1. Multi-Source Aggregation**

```python
import requests
from typing import Dict, List

class FoodDataAggregator:
    def __init__(self):
        self.nutritionix_key = os.getenv("NUTRITIONIX_KEY")
        self.usda_key = os.getenv("USDA_KEY")
        self.cache = {}  # Redis in production
    
    def get_food_data_by_barcode(self, barcode: str) -> Dict:
        """Aggregate from multiple sources"""
        
        # Priority 1: Check cache
        if barcode in self.cache:
            return self.cache[barcode]
        
        # Priority 2: Nutritionix (best for barcodes)
        try:
            nutritionix_data = self._fetch_nutritionix(barcode)
            if nutritionix_data:
                aggregated = nutritionix_data
        except:
            aggregated = {}
        
        # Priority 3: USDA (fallback for nutrition)
        try:
            usda_data = self._fetch_usda(barcode)
            aggregated = self._merge_data(aggregated, usda_data)
        except:
            pass
        
        # Priority 4: Open Food Facts (verification)
        try:
            off_data = self._fetch_open_food_facts(barcode)
            aggregated = self._verify_and_merge(aggregated, off_data)
        except:
            pass
        
        # Priority 5: Check brand website
        try:
            brand_data = self._scrape_brand_website(aggregated.get("brand"), aggregated.get("name"))
            aggregated = self._merge_with_priority(aggregated, brand_data)
        except:
            pass
        
        # Cache result
        self.cache[barcode] = aggregated
        return aggregated
    
    def _fetch_nutritionix(self, barcode: str) -> Dict:
        """Nutritionix API call"""
        headers = {
            "x-app-id": "YOUR_APP_ID",
            "x-app-key": "YOUR_APP_KEY"
        }
        url = "https://trackapi.nutritionix.com/v2/search/item"
        params = {"upc": barcode}
        
        response = requests.get(url, headers=headers, params=params)
        if response.status_code == 200:
            item = response.json()["items"][0]
            return {
                "name": item["item_name"],
                "brand": item["brand_name"],
                "calories": item["nf_calories"],
                "protein": item["nf_total_protein"],
                "carbs": item["nf_total_carbohydrate"],
                "fat": item["nf_total_fat"],
                "sodium": item.get("nf_sodium_mg", 0),
                "sugar": item.get("nf_sugars_g", 0),
                "source": "nutritionix"
            }
        return {}
    
    def _fetch_usda(self, product_name: str) -> Dict:
        """USDA FoodData Central API"""
        url = "https://api.nal.usda.gov/fdc/v1/foods/search"
        params = {
            "query": product_name,
            "api_key": self.usda_key,
            "pageSize": 1
        }
        
        response = requests.get(url, params=params)
        if response.status_code == 200:
            foods = response.json()["foods"]
            if foods:
                food = foods[0]
                nutrients = {n["nutrientName"]: n.get("value", 0) for n in food.get("foodNutrients", [])}
                return {
                    "name": food["description"],
                    "calories": nutrients.get("Energy", 0),
                    "protein": nutrients.get("Protein", 0),
                    "carbs": nutrients.get("Carbohydrate", 0),
                    "fat": nutrients.get("Total lipid", 0),
                    "source": "usda"
                }
        return {}
    
    def _fetch_open_food_facts(self, barcode: str) -> Dict:
        """Open Food Facts API"""
        url = f"https://world.openfoodfacts.org/api/v0/product/{barcode}.json"
        
        response = requests.get(url)
        if response.status_code == 200:
            product = response.json()["product"]
            return {
                "name": product.get("product_name"),
                "brand": product.get("brands"),
                "ingredients": product.get("ingredients_text"),
                "allergens": product.get("allergens"),
                "source": "open_food_facts"
            }
        return {}
    
    def _verify_and_merge(self, primary: Dict, secondary: Dict) -> Dict:
        """
        Verify consistency between sources
        Alert if nutritional values differ by >10%
        """
        if not secondary.get("calories"):
            return primary
        
        primary_cal = primary.get("calories", 0)
        secondary_cal = secondary.get("calories", 0)
        
        if primary_cal and secondary_cal:
            difference = abs(primary_cal - secondary_cal) / primary_cal * 100
            if difference > 10:
                # Log discrepancy for human review
                print(f"⚠️ Calorie mismatch: {primary_cal} vs {secondary_cal} ({difference}%)")
                primary["discrepancy_flag"] = True
        
        # Merge only unique fields from secondary
        for key, value in secondary.items():
            if key not in primary and value:
                primary[key] = value
        
        return primary
```

```python
            "is_valid": True,
            "warnings": [],
            "errors": []
        }
        
        # Check 1: Sodium limits (FSSAI max 400mg/serving)

                f"High sodium content: {product['sodium']}mg (limit: 400mg)"

            validation_report["warnings"].append(
                f"High sugar content: {product['sugar']}g (recommended: <25g)"
            )
        
        # Check 3: Trans fat (FSSAI ban)
        if product.get("trans_fat", 0) > 0:
            validation_report["errors"].append(

            validation_report["is_valid"] = False
```
